# Mutation for register
mutation {
  register(input: {
    email: "daniel@studybuddies.com",
    password: "Daniel123!",
    username: "danielnedelcu",
    firstName: "Daniel",
    lastName: "Nedelcu"
  }) {
    token
    user { userID email username }
  }
}


# Mutation for login
mutation Login($input: LoginInput!) {
  login(input: $input) {
    token
    user {
      userID
      email
      username
    }
  }
}
